int strLenCmp(char s[],char t[]){
	int i, j, r;
	for(i = 0; s[i] != '\0'; i++){
	}
	for(j = 0; t[j] != '\0'; j++){
	}
	if(i == j){
		r = 0;
	}else if (i > j){

	}
